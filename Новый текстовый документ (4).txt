#include <iostream>
using namespace std;
int main()
{
	setlocale(LC_ALL, "Rus");
	const int n = 13;
	int mas[n];
	for (int i = 0;i < n;i++)
	{
		cin >> mas[i];
	}
	int sum=0;
	for (int i = 0; i < n;i++)
	{
		sum = sum + mas[i];
	}
	cout << "всего свободных гигобайтов :" << sum << endl;
	int s;
	cout << "сколько весит фильм" << endl;
	cin >> s;
	int a=0;
	for (int i = 0;i < n;i++)
	{
		if (mas[i] >= s)
		{
			a++;	
		}
	}  
	int g=mas[0]; 
	int u = 0;
	for (int i = 0;i < n;i++)
	{
		if (mas[i]>g)
		{
			g = mas[i];
			u = i;
		}
	}
	int p = mas[0];
	int l = 0;
	for (int i = 0;i < n;i++)
	{
		if (mas[i] < p)
		{
			p = mas[i];
			l = i;
		}
	}
	for (int i = 0;i < n;i++)
	{
		if (mas[i] < 10)
		{
			mas[i] = mas[i] + 10;
		}
	}
	cout << "минимальная память" << p << endl;
	cout << "компьютер с минимальной памятью" << l << endl;
	cout <<"компьютер с максимальной памятью"<< u << endl;
	cout << "всего можнет потинуть компов :" << a << endl;
	cout << "максимальная память :" << g << endl;
	for (int i = 0;i < n;i++)
	{
		cout << mas[i] << "  " << endl;
	}
	system("pause");
}
